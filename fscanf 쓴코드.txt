#include <stdio.h>
int main(void) {
	FILE* file = fopen("input.txt", "r");
	int upper = 0, lower = 0, arabic = 0, blank = 0;
	int chtread = 0;
	if (file == '\0') {
		return 1;
	}
	while (!feof(file)) {
		fscanf(file, "%c", &chtread);
		if (chtread >= '0' && chtread <= '9')
			arabic++;
		else if (chtread >= 'A' && chtread <= 'Z')
			upper++;
		else if (chtread >= 'a' && chtread <= 'z')
			lower++;
		else if (chtread == ' ')
			blank++;
	}
	fprintf(stdout, "upper: %d\nlower: %d\narabic: %d\nblank: %d", upper, lower, arabic, blank);
	fclose(file);
}
